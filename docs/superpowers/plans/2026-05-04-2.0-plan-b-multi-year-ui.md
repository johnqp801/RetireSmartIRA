# V2.0 Plan B — Multi-Year Tax Strategy UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Spec:** [docs/superpowers/specs/2026-05-04-2.0-plan-b-multi-year-ui-design.md](../specs/2026-05-04-2.0-plan-b-multi-year-ui-design.md)

**Goal:** Build the V2.0 Multi-Year Tax Strategy UI surface — a single consolidated "Tax Planning" tab that replaces Scenarios + Tax Summary, hosting the engine's multi-year output with a master-detail layout (iPad/macOS) or bottom-sheet (iPhone/compact), plus first-run migration UX for 1.8 → 2.0 users.

**Architecture:** New `MultiYearStrategyManager: ObservableObject` parallel to existing `DataManager` and `ScenarioStateManager`. Decoupled recompute: right-pane (Year 1 detail) reads DataManager properties synchronously for instant feedback; macro pane (waterfall, hero, year list) is debounced 500ms after slider release, runs the engine on a background `Task`, surfaces results via `@Published` properties. Two-cache strategy with dirty-flag pattern guarantees the engine-optimal baseline stays correct under cancellation.

**Tech Stack:** Swift 5.9+, SwiftUI, Combine, Swift Charts, XCTest. Xcode project (not SPM Package). Native macOS + iOS/iPadOS targets.

**Branch strategy:** All work lands on `2.0/plan-b-ui` branched from `2.0/multi-year-engine` (NOT `main` — engine work is on the 2.0 branch). Final merge to `main` happens as the single 2.0 release branch.

**Reading order for the engineer:** §1 (header, this section) → §2 (file structure) → §3 (Phase 0 standalone rename) → §4+ (subsequent phases). Each phase is self-contained; do not skip ahead.

**Test framework:** XCTest with `@testable import RetireSmartIRA`. Snapshot tests extend the 1.9 pattern. Build/test command: `xcodebuild -project RetireSmartIRA.xcodeproj -scheme RetireSmartIRA test -destination 'platform=iOS Simulator,name=iPhone 16'` (also run with `-destination 'platform=macOS'`).

func recompute(reason: RecomputeReason) {
    // Mark optimal-cache dirty if assumptions or static inputs changed.
    // The flag survives task cancellation — see spec §5.2 dirty-flag pattern.
    if reason == .assumptionsChanged || reason == .appLaunch {
        needsOptimalRecompute = true
    }

    debounceTask?.cancel()
    isComputing = true

    debounceTask = Task { [weak self] in
        // Debounce window — 500ms after last call.
        try? await Task.sleep(nanoseconds: 500_000_000)
        guard !Task.isCancelled, let self else { return }

        guard let dataManager = self.dataManager,
              let scenarioStateManager = self.scenarioStateManager else {
            // Without managers, can't build inputs. Surface error in Task 8 phase.
            await MainActor.run { self.isComputing = false }
            return
        }

        // Build static inputs snapshot (this read is @MainActor-bound).
        let assumptions = self.assumptions
        let staticInputs = MultiYearInputAdapter.build(
            from: dataManager,
            scenarioState: scenarioStateManager,
            assumptions: assumptions
        )

        // Run engine off-main. Snapshot dirty flag before computing.
        let needsOptimal = self.needsOptimalRecompute
        let result: (optimal: MultiYearStrategyResult?, current: MultiYearStrategyResult) = await Task.detached(priority: .userInitiated) {
            let engine = MultiYearTaxStrategyEngine()
            let current = engine.compute(inputs: staticInputs, assumptions: assumptions)
            let optimal: MultiYearStrategyResult? = needsOptimal
                ? engine.compute(
                    inputs: Self.staticInputsWithoutOverrides(staticInputs, dataManager: dataManager),
                    assumptions: assumptions
                  )
                : nil
            return (optimal, current)
        }.value

        guard !Task.isCancelled else { return }

        await MainActor.run {
            if let optimal = result.optimal {
                self.engineOptimalResult = optimal
                self.needsOptimalRecompute = false  // clear ONLY on success
            }
            self.currentResult = result.current
            self.hasEverComputed = true
            self.isComputing = false
        }
    }
}

//
//  StrategySummarySynthesizer.swift
//  RetireSmartIRA
//
//  Pure-logic helper: turns a recommendedPath + tradeOffs into 2-3
//  plain-English sentences describing the engine's strategy.
//

import Foundation

enum StrategySummarySynthesizer {

    static func synthesize(
        path: [YearRecommendation],
        tradeOffs: [ConstraintHit]
    ) -> String {
        guard !path.isEmpty else {
            return "No multi-year strategy computed yet."
        }

        var sentences: [String] = []

        // 1. Roth conversion cluster
        let rothCluster = clusterRothConversions(in: path)
        if let clause = rothCluster {
            sentences.append(clause)
        }

        // 2. SS claim mention
        if let ssClause = describeSSClaims(in: path) {
            sentences.append(ssClause)
        }

        // 3. RMD-fill mention (if any year shows trad-withdrawal triggered by RMD age)
        if let rmdClause = describeRMDPhase(in: path) {
            sentences.append(rmdClause)
        }

        // 4. Trade-offs accepted summary
        if !tradeOffs.isEmpty {
            sentences.append(describeTradeOffs(tradeOffs))
        }

        return sentences.joined(separator: " ")
    }

    private static func clusterRothConversions(in path: [YearRecommendation]) -> String? {
        let rothYears: [(year: Int, amount: Double)] = path.compactMap { y in
            for action in y.actions {
                if case .rothConversion(let amount) = action, amount > 0 {
                    return (y.year, amount)
                }
            }
            return nil
        }
        guard !rothYears.isEmpty else { return nil }

        let first = rothYears.first!.year
        let last = rothYears.last!.year
        let amounts = rothYears.map { $0.amount }
        let minAmt = (amounts.min() ?? 0) / 1000
        let maxAmt = (amounts.max() ?? 0) / 1000

        let amtPhrase: String
        if minAmt == maxAmt {
            amtPhrase = "$\(Int(minAmt))K/yr"
        } else {
            amtPhrase = "$\(Int(minAmt))–\(Int(maxAmt))K/yr"
        }

        let rangePhrase: String
        if first == last {
            rangePhrase = "in \(first)"
        } else {
            rangePhrase = "from \(first) through \(last)"
        }

        return "Convert \(amtPhrase) to Roth \(rangePhrase)."
    }

    private static func describeSSClaims(in path: [YearRecommendation]) -> String? {
        var primaryYear: Int?
        var spouseYear: Int?
        for y in path {
            for action in y.actions {
                if case .claimSocialSecurity(let spouse) = action {
                    if spouse == .primary { primaryYear = y.year }
                    if spouse == .spouse { spouseYear = y.year }
                }
            }
        }
        switch (primaryYear, spouseYear) {
        case (.some(let p), .some(let s)) where p == s:
            return "Both spouses claim SS in \(p)."
        case (.some(let p), .some(let s)):
            return "Claim SS in \(p) (you) and \(s) (spouse)."
        case (.some(let p), nil):
            return "Claim SS in \(p)."
        case (nil, .some(let s)):
            return "Spouse claims SS in \(s)."
        case (nil, nil):
            return nil
        }
    }

    private static func describeRMDPhase(in path: [YearRecommendation]) -> String? {
        // Heuristic: first year where traditionalWithdrawal action appears AND
        // the year >= 73 (RMD age) is treated as RMD start.
        for y in path {
            for action in y.actions {
                if case .traditionalWithdrawal = action {
                    return "RMD-fill the bracket starting in \(y.year)."
                }
            }
        }
        return nil
    }

    private static func describeTradeOffs(_ hits: [ConstraintHit]) -> String {
        let count = hits.count
        if count == 1 {
            let h = hits[0]
            switch h.type {
            case .irmaaTier(let level):
                return "Accepts 1 IRMAA Tier \(level) hit at \(h.year)."
            case .acaCliff:
                return "Accepts 1 ACA cliff hit at \(h.year)."
            case .bracketOverrun(let from, let to):
                return "Accepts 1 \(from)% → \(to)% bracket overrun at \(h.year)."
            }
        }
        return "Accepts \(count) trade-offs (see Trade-offs accepted card for detail)."
    }
}

import Foundation

enum YearOverYearDeltaSynthesizer {

    struct Result {
        let taxDelta: Double?
        let causeSentence: String?
    }

    static func synthesize(prior: YearRecommendation?, current: YearRecommendation) -> Result {
        guard let prior = prior else {
            return Result(taxDelta: nil, causeSentence: nil)
        }

        let delta = current.taxBreakdown.total - prior.taxBreakdown.total

        // Cause sentence priority: RMD start > SS claim > IRMAA cross > bracket change > none
        if hasNewRMDWithdrawal(prior: prior, current: current) {
            return Result(taxDelta: delta, causeSentence: "RMDs begin.")
        }
        if hasNewSSClaim(prior: prior, current: current) {
            return Result(taxDelta: delta, causeSentence: "Social Security claimed.")
        }
        if let tier = irmaaCrossed(prior: prior, current: current) {
            return Result(taxDelta: delta, causeSentence: "IRMAA Tier \(tier) crossed.")
        }

        // No notable cause for small deltas
        if abs(delta) < 1_000 {
            return Result(taxDelta: delta, causeSentence: nil)
        }

        return Result(taxDelta: delta, causeSentence: "Bracket / income change.")
    }

    private static func hasNewRMDWithdrawal(prior: YearRecommendation, current: YearRecommendation) -> Bool {
        let priorHas = prior.actions.contains { if case .traditionalWithdrawal = $0 { return true } else { return false } }
        let currentHas = current.actions.contains { if case .traditionalWithdrawal = $0 { return true } else { return false } }
        return !priorHas && currentHas
    }

    private static func hasNewSSClaim(prior: YearRecommendation, current: YearRecommendation) -> Bool {
        let priorHas = prior.actions.contains { if case .claimSocialSecurity = $0 { return true } else { return false } }
        let currentHas = current.actions.contains { if case .claimSocialSecurity = $0 { return true } else { return false } }
        return !priorHas && currentHas
    }

    private static func irmaaCrossed(prior: YearRecommendation, current: YearRecommendation) -> Int? {
        // If IRMAA went from 0 to non-zero, derive tier from MAGI vs CMS thresholds.
        // For V2.0 simplification, just say "IRMAA tier crossed" without numeric tier.
        if prior.taxBreakdown.irmaa == 0 && current.taxBreakdown.irmaa > 0 {
            return 1  // placeholder — real tier requires CMS threshold lookup
        }
        return nil
    }
}

            return "Different from engine"
        }
        if useNeutralFraming, case .significantlyOffPlan = state {
            return "Different from engine"
        }
        switch state {
        case .onPlan: return "On plan"
        case .nearOptimal(let delta): return "Near optimal (\(formatDelta(delta)))"
        case .offPlan(let delta): return "Off plan: \(formatDelta(delta))"
        case .significantlyOffPlan(let delta): return "Significantly off plan: \(formatDelta(delta))"
        }
    }

    private var showsResetLink: Bool {
        if useNeutralFraming { return false }
        switch state {
        case .offPlan, .significantlyOffPlan: return true
        default: return false
        }

    private var showsSeeEnginePlanLink: Bool {
        guard useNeutralFraming else { return false }
        switch state {
        case .offPlan, .significantlyOffPlan: return true
        default: return false
        }
    }

    private func formatDelta(_ d: Double) -> String {
        let sign = d < 0 ? "-" : "+"
        let abs = Int(d.magnitude.rounded())
        return "\(sign)$\(abs.formatted(.number))"

extension OffPlanIndicator.State {
    /// Construct from lifetime tax delta (currentResult - engineOptimalResult).
    /// Positive delta means user's plan costs MORE than optimal (worse).
    static func fromDelta(_ delta: Double, thresholds: Thresholds = .default) -> Self {
        let abs = delta.magnitude
        if abs < thresholds.nearOptimal {
            return delta == 0 ? .onPlan : .nearOptimal(deltaDollars: -delta)
        }
        if abs < thresholds.major {
            return .offPlan(deltaDollars: -delta)
        }
        return .significantlyOffPlan(deltaDollars: -delta)
    }

    struct Thresholds {
        let nearOptimal: Double  // < this = near optimal
        let major: Double         // >= this = significantly off
        static let `default` = Thresholds(nearOptimal: 1_000, major: 25_000)
    }
