### 2.1 `AccountSnapshot` (replaces original Task 1.2 definition)

Adds `taxable` and `hsa` fields. These are NOT new `AccountType` cases; they're tracked as separate fields in the engine's snapshot model only.

```swift
public struct AccountSnapshot: Codable, Equatable {
    public let traditional: Double  // sum of trad IRA + trad 401k + inherited trad IRA
    public let roth: Double          // sum of roth IRA + roth 401k + inherited roth IRA
    public let taxable: Double       // user-input balance, not modeled as AccountType in 1.9
    public let hsa: Double           // user-input balance, not modeled as AccountType in 1.9

    public init(traditional: Double, roth: Double, taxable: Double, hsa: Double) {
        self.traditional = traditional
        self.roth = roth
        self.taxable = taxable
        self.hsa = hsa
    }

    public var total: Double { traditional + roth + taxable + hsa }
    public static let zero = AccountSnapshot(traditional: 0, roth: 0, taxable: 0, hsa: 0)
}
```

The engine collapses 1.9's 6 retirement-account types into 2 buckets (`traditional`, `roth`) for optimization purposes. The adapter (Task 1.0 below) does the rollup.

```swift
//
//  MultiYearInputAdapter.swift
//  RetireSmartIRA
//
//  Builds MultiYearStaticInputs from the runtime ObservableObject state
//  (DataManager + ScenarioStateManager) for consumption by the pure-calc
//  Multi-Year Tax Strategy engine.
//

import Foundation

@MainActor
public enum MultiYearInputAdapter {

    public static func build(
        from dataManager: DataManager,
        scenarioState: ScenarioStateManager,
        assumptions: MultiYearAssumptions
    ) -> MultiYearStaticInputs {

        // Roll up the 1.9 AccountType cases into the engine's two buckets.
        // (Account access patterns: see AccountsManager.swift for the actual API —
        // likely something like dataManager.accountsManager.accounts: [Account].)
        let allAccounts = dataManager.accountsManager.accounts  // adapt to real API
        let traditional = allAccounts
            .filter { [.traditionalIRA, .traditional401k, .inheritedTraditionalIRA].contains($0.accountType) }
            .reduce(0.0) { $0 + $1.balance }
        let roth = allAccounts
            .filter { [.rothIRA, .roth401k, .inheritedRothIRA].contains($0.accountType) }
            .reduce(0.0) { $0 + $1.balance }

        let snapshot = AccountSnapshot(
            traditional: traditional,
            roth: roth,
            taxable: assumptions.currentTaxableBalance,
            hsa: assumptions.currentHSABalance
        )

        // Demographics — adapt field accessors to real DataManager shape
        let primaryAge = dataManager.currentYear - dataManager.yourBirthYear
        let spouseAge: Int? = dataManager.enableSpouse ? (dataManager.currentYear - dataManager.spouseBirthYear) : nil

        // SS — adapt to actual DataManager+SocialSecurity field names
        let primaryClaimAge = dataManager.yourSSClaimAge
        let spouseClaimAge: Int? = dataManager.enableSpouse ? dataManager.spouseSSClaimAge : nil

        // Pension / wage — adapt to actual income source iteration
        let primaryPension = dataManager.incomeSourcesManager.totalAnnualPension(for: .primary)
        let spousePension = dataManager.enableSpouse ? dataManager.incomeSourcesManager.totalAnnualPension(for: .spouse) : 0.0
        let primaryWage = dataManager.yourWageIncome
        let spouseWage = dataManager.enableSpouse ? dataManager.spouseWageIncome : 0.0

        // ACA / Medicare
        let acaEnrolled = scenarioState.enableACAModeling
        let acaSize = scenarioState.acaHouseholdSize
        let primaryMedAge = 65  // TODO: derive from scenarioState.yourMedicarePlanType timing if needed
        let spouseMedAge: Int? = dataManager.enableSpouse ? 65 : nil

        // Expenses — pull from existing 1.9 baseline (probably DataManager.annualExpenses or similar)
        let baselineExpenses = dataManager.annualExpenses

        return MultiYearStaticInputs(
            startingBalances: snapshot,
            primaryCurrentAge: primaryAge,
            spouseCurrentAge: spouseAge,
            filingStatus: dataManager.filingStatus,
            state: dataManager.state,
            primarySSClaimAge: primaryClaimAge,
            spouseSSClaimAge: spouseClaimAge,
            primaryExpectedBenefitAtFRA: dataManager.primarySSBenefit,
            spouseExpectedBenefitAtFRA: dataManager.enableSpouse ? dataManager.spouseSSBenefit : nil,
            primaryBirthYear: dataManager.yourBirthYear,
            spouseBirthYear: dataManager.enableSpouse ? dataManager.spouseBirthYear : nil,
            primaryWageIncome: primaryWage,
            spouseWageIncome: spouseWage,
            primaryPensionIncome: primaryPension,
            spousePensionIncome: spousePension,
            acaEnrolled: acaEnrolled,
            acaHouseholdSize: acaSize,
            primaryMedicareEnrollmentAge: primaryMedAge,
            spouseMedicareEnrollmentAge: spouseMedAge,
            baselineAnnualExpenses: baselineExpenses
        )
    }
}
```

```swift
public struct OptimizationEngine {
    public init() {}

    public struct Result {
        public let recommendedPath: [YearRecommendation]
        public let tradeOffsAccepted: [ConstraintHit]
    }

    public func optimize(
        inputs: MultiYearStaticInputs,
        assumptions: MultiYearAssumptions
    ) -> Result {
        // 1. Allocate dp[year][bracket][irmaa][rothBucket] table (Double, Double.infinity initial)
        // 2. Backward induction: for each state, enumerate candidate Roth conversion amounts
        //    (5-10 levels), compute year-cost via ProjectionEngine, set dp[s] = min(dp[s], cost + dp[next])
        // 3. Forward path reconstruction: from start state, follow argmin actions year-by-year
        // 4. Detect ConstraintHits along the chosen path via ConstraintAcceptor
        // 5. Return Result
    }
}
```

State-space discretization (per Task 0.5 decision):
- Year: 30
- Bracket bucket: 7 (federal 2026 brackets 10/12/22/24/32/35/37%)
- IRMAA tier: 6 (tier 0–5)
- Roth balance bucket: 20 (log-uniform from 0 to scenario start × 5)

Action enumeration per state: 5–10 candidate Roth conversion amounts. Withdrawals + contributions (HSA, 401k) are NOT enumerated; they're auto-determined by `withdrawalOrderingRule` + scenario inputs within `ProjectionEngine`.

## 6. Open: persistence of `MultiYearAssumptions`

The original Task 1.6 also covered persistence (Codable round-trip of the new field on `Scenario`). Since there's no `Scenario` struct, persistence needs a different design:

**Two viable options:**

A. **Add `MultiYearAssumptions` fields to `ScenarioStateManager`** as `@Published` properties, persisted by `PersistenceManager`'s existing scenario-state save path. Most consistent with 1.9 conventions.

B. **Persist `MultiYearAssumptions` as a separate document** (e.g., `multiYearAssumptions.json` alongside the scenario state file). Cleaner separation but adds a second persistence path.

**Defer this decision to Plan B (UI).** The engine itself is stateless and consumes `MultiYearAssumptions` as a value. Where the UI gets that value from is a UI/state concern. For Phase 1 (engine implementation), tests construct `MultiYearAssumptions` directly — no persistence needed.
