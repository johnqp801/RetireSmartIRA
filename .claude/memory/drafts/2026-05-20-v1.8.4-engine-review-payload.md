### Engine code — `RetirementIncomeExemptions` schema

```swift
struct RetirementIncomeExemptions {
    /// 42 states exempt SS for state income tax
    var socialSecurityExempt: Bool = true

    /// Whether pension income is exempt from state income tax
    var pensionExemption: ExemptionLevel = .none

    /// Whether IRA/401(k) withdrawals are exempt
    var iraWithdrawalExemption: ExemptionLevel = .none

    /// When true AND filing status is MFJ AND both spouses are 59½+,
    /// the engine doubles the partial cap (each spouse gets their own).
    var exemptionAppliesPerIndividual: Bool = false

    /// Minimum age for the regular pension/IRA exemption to apply.
    /// When 0 (default), no age gate. Setting >0 also enables earlyAgeTier
    /// fallback for younger ages.
    var regularExemptionMinAge: Int = 0

    /// Optional reduced exemption for an early-age tier (e.g. GA 62-64).
    /// Applied when user's age is in this range AND age < regularExemptionMinAge.
    var earlyAgeTier: AgeTier? = nil

    /// When true, pension AND IRA share a single combined cap (e.g. CO).
    /// Engine uses pensionExemption.partial.maxExempt as the shared cap.
    var pensionAndIRAShareSingleCap: Bool = false

    var capitalGainsTreatment: CapGainsTreatment = .followsFederal

    struct AgeTier {
        let ageRange: ClosedRange<Int>
        let level: ExemptionLevel
    }

    enum ExemptionLevel {
        case none
        case full
        case partial(maxExempt: Double)
    }
}
```

### Engine code — `applyRetirementExemptions` (the consumer)

```swift
static func applyRetirementExemptions(
    income: Double,
    config: StateTaxConfig,
    state: USState,
    taxableSocialSecurity: Double,
    incomeSources: [IncomeSource],
    primaryAge: Int,
    spouseAge: Int,
    enableSpouse: Bool,
    scenarioRetirementDistributions: Double = 0,
    scenarioRothConversionAmount: Double = 0
) -> Double {
    var adjusted = income
    let exemptions = config.retirementExemptions

    if exemptions.socialSecurityExempt {
        adjusted -= taxableSocialSecurity
    }

    // Per-individual cap multiplier: MFJ where BOTH spouses 59½+ AND
    // state's exemption is per-individual → cap doubles.
    let bothSpouses59Plus = enableSpouse && primaryAge >= 59 && spouseAge >= 59
    let perIndividualMultiplier: Double =
        (exemptions.exemptionAppliesPerIndividual && bothSpouses59Plus) ? 2.0 : 1.0

    // Resolve effective exemption level given age:
    //   1) age >= regularExemptionMinAge → regular exemption
    //   2) age in earlyAgeTier.ageRange  → tier's level
    //   3) else                          → .none
    // When regularExemptionMinAge == 0, no age gate; regular applies.
    let effectiveAge = enableSpouse ? max(primaryAge, spouseAge) : primaryAge
    let minAge = exemptions.regularExemptionMinAge

    func resolveLevel(regular: RetirementIncomeExemptions.ExemptionLevel) -> RetirementIncomeExemptions.ExemptionLevel {
        if minAge > 0 {
            if effectiveAge >= minAge {
                return regular
            }
            if let tier = exemptions.earlyAgeTier, tier.ageRange.contains(effectiveAge) {
                return tier.level
            }
            return .none
        }
        return regular
    }

    let effectivePensionExemption = resolveLevel(regular: exemptions.pensionExemption)
    let effectiveIRAExemption = resolveLevel(regular: exemptions.iraWithdrawalExemption)

    let pensionIncome = incomeSources.filter { $0.type == .pension }
                                      .reduce(0) { $0 + $1.annualAmount }

    // IRA-side income: explicit .rmd rows + scenario-level distributions
    // (RMDs + extra withdrawals + inherited-IRA RMDs). Scenario portion
    // is age-gated at 59½ (early withdrawals are taxable in most states).
    let rmdSourceIncome = incomeSources.filter { $0.type == .rmd }
                                        .reduce(0) { $0 + $1.annualAmount }
    let retirementAge = primaryAge >= 59 || (enableSpouse && spouseAge >= 59)
    let scenarioExemptable = retirementAge ? scenarioRetirementDistributions : 0
    let iraIncome = rmdSourceIncome + scenarioExemptable

    if exemptions.pensionAndIRAShareSingleCap {
        let combinedIncome = pensionIncome + iraIncome
        switch effectivePensionExemption {
        case .full:
            adjusted -= combinedIncome
        case .partial(let maxExempt):
            adjusted -= min(combinedIncome, maxExempt * perIndividualMultiplier)
        case .none:
            break
        }
    } else {
        switch effectivePensionExemption {
        case .full:
            adjusted -= pensionIncome
        case .partial(let maxExempt):
            adjusted -= min(pensionIncome, maxExempt * perIndividualMultiplier)
        case .none:
            break
        }
        switch effectiveIRAExemption {
        case .full:
            adjusted -= iraIncome
        case .partial(let maxExempt):
            adjusted -= min(iraIncome, maxExempt * perIndividualMultiplier)
        case .none:
            break
        }
    }

    return adjusted
}
```

### Primary-source quotes for each state

#### Maryland — Comptroller of Maryland Technical Bulletin No. 51 (effective April 10, 2025)

> **F. Pension exclusion.**
> Maryland's maximum pension exclusion of **$39,500** is available to taxpayers if:
> 1) On the last day of the tax year:
>    - the taxpayer was 65 or older, **or**
>    - Taxpayer is totally disabled; or
>    - Taxpayer's spouse was totally disabled;
>
> AND
>
> 2) The taxpayer included taxable income received as a pension, annuity, or endowment from an "employee retirement system" qualified under Sections 401(a), 403 or 457(b) of the Internal Revenue Code on their federal return. **A traditional IRA, a Roth IRA, a rollover IRA, a simplified employee plan (SEP), a Keogh plan, an ineligible deferred compensation plan, or foreign retirement income does not qualify.**

Statute: Md Tax-General § 10-209.

#### New York — NY Tax Law § 612(c)(3-a) (Pensions and annuities $20,000 exclusion)

> "Pensions and annuities received by an individual who has attained the age of fifty-nine and one-half"

Per IT-201 instructions (NYS DOR): the $20,000 exclusion is applied **per qualifying individual** on the return. On a joint return, each spouse who separately meets the age and qualifying-income tests gets a separate $20,000 exclusion (combined household max = $40,000 when both qualify).

#### Georgia — O.C.G.A. § 48-7-27(a)(5) (Georgia Retirement Income Exclusion)

Effective for tax years beginning on or after January 1, 2012:
- A taxpayer aged **62 to less than 65** during any part of the taxable year may exclude up to **$35,000** of retirement income.
- A taxpayer aged **65 or older** during any part of the taxable year may exclude up to **$65,000** of retirement income.

Verified from Justia (Georgia Code § 48-7-27 (2024)) and the Georgia DOR Retirement Income Exclusion page. The exclusion is **per qualifying individual** — each spouse separately meeting the age test gets their own exclusion amount.

#### New Jersey — NJSA 54A:6-15 (NJ Pension and Other Retirement Income Exclusion)

> Eligibility requires the taxpayer to be **62 years of age or older** OR totally disabled.

For tax years beginning on or after January 1, 2020, maximum exclusion amounts:
- MFJ: up to $100,000
- Single: up to $75,000
- MFS: up to $50,000

AGI-based phaseout (tax years beginning 2021+): full exclusion below $100K AGI; reduced 50%/37.5%/25% in $100K–$125K tier; reduced 25%/18.75%/12.5% in $125K–$150K tier; zero above $150K.

(**The code's known approximations of filing-status-aware caps and AGI phaseout are explicitly OUT OF SCOPE for this review.** Only review whether the age-62-minimum gate is correctly applied.)

#### Colorado — C.R.S. § 39-22-104(4)(f) (Pension and Annuity Subtraction)

Per Colorado DOR guidance ("Income Tax Topics: Social Security, Pensions and Annuities"):

> Amounts subtracted under C.R.S. § 39-22-104(4)(f) are capped at **$20,000 per tax year for individuals age 55-64**, and at **$24,000 per tax year for any individual who is 65 years of age or older.**

The subtraction applies to "amounts received as pensions and annuities," which per CO DOR guidance **includes IRA distributions counting against the same annual cap** (single combined cap, NOT separate caps for pension and IRA).

Note: SB25-136 (which would have removed the cap entirely for TY2026) was **Postponed Indefinitely on 02/27/2025** — did NOT become law. The $20K / $24K caps remain in effect.
